#version 410 core
layout (location = 0) in vec3 vertices_pos;
layout (location = 2) in vec3 color;
layout (location = 1) in vec2 texCoord;

out vec4 f_color;
out vec2 f_texCoord;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

void main() {
    gl_Position = projection * view * model * vec4(vertices_pos, 1.0);
    f_color = vec4(1.0f, 0.0f, 0.0f, 1.0f);
    f_texCoord = texCoord;
}
